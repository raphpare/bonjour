# Month View

## Examples

### HTML

```HTML
<!-- Month View -->
<div id="calendarMonthView"></div>


<!-- Your UI controls -->
<button id="btnToday">Today</button>
<button id="btnPrevious">Previous</button>
<button id="btnNext">Next</button>
<button id="btnAdd">Add event</button>
```

### TypeScript

```TS
import { B5rMonthView } from 'b5r-bonjour/lib/month-view';

const CalendarMonthView = new B5rMonthView(
    document.getElementById('calendarMonthView')
);

document
    .getElementById('btnToday')
    .addEventListener('click', () => {
        CalendarMonthView.today()
    });


document
    .getElementById('btnPrevious')
    .addEventListener('click', () => {
        CalendarMonthView.previous()
    });

document
    .getElementById('btnNext')
    .addEventListener('click', () => {
        CalendarMonthView.next()
    });

document.getElementById('btnAdd').addEventListener('click', () => {
    CalendarMonthView.setEvents([
        {
            id: '1',
            title: 'Event 1',
            subtitle: 'Subtitle',
            dateRange: {
                start: new Date(),
                end: new Date(new Date().setHours(new Date().getHours() + 2)),
            },
        },
    ]);
});
```
